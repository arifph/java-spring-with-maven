```sql
join user_roles ur
on mr.ROLES_ID = ur.ROLE_ID
join menu mn
on mr.APPLICATIONMENU_ID = mn.ID
```

```java
public List<String> getMenuUniqueByUser(String username) {
	// TODO Auto-generated method stub
	Session session = sessionFactory.getCurrentSession();
	String sql = "select distinct(mn.url) from menu_role mr\r\n" + 
			"join role rl \r\n" + 
			"on mr.ROLES_ID = rl.ID\r\n" + 
			"join user_roles ur \r\n" + 
			"on mr.ROLES_ID = ur.ROLE_ID\r\n" + 
			"join menu mn \r\n" + 
			"on mr.APPLICATIONMENU_ID = mn.ID\r\n" + 
			"join users us\r\n" + 
			"on ur.USER_ID = us.ID\r\n" + 
			"where us.USERNAME =:username";

	Query query = session.createSQLQuery(sql);
	query.setParameter("username", username);
	List<String> result = query.list();
	if(!result.isEmpty()) {
		return result;
	}
	return new ArrayList<String>();
}
```

## Filter Menu ControllerAdvice
```java
@ControllerAdvice
public class AuthMenu {	
	@Autowired
	HttpServletRequest request;
	@Autowired
	HttpServletResponse response;
	public static boolean REQUEST_IS_ACCEPTENCE = false;
	
	@ModelAttribute
	public void auth(Authentication auth) {
		System.out.println("current url : "+ request.getRequestURI());
		if(auth != null) {
			UserDetails userDetails = (UserDetails) auth.getPrincipal();
			HttpSession theSession = request.getSession();
			User user = (User)theSession.getAttribute("application-user");
			List<String> menus = user.getMenusTemp();
			
			for(String menu : menus) {
				if(menu.equalsIgnoreCase(request.getRequestURI())) {
					REQUEST_IS_ACCEPTENCE = true;
				} 
			}
			if(request.getRequestURI().equalsIgnoreCase("/dashboard")) {
				REQUEST_IS_ACCEPTENCE = true;
			}
			
			if(request.getRequestURI().equalsIgnoreCase("/login")) {
				REQUEST_IS_ACCEPTENCE = true;
			}
			
			if(!REQUEST_IS_ACCEPTENCE) {
				throw new AccessDeniedException("403 Access Denied");
			}
			
			REQUEST_IS_ACCEPTENCE = false;
		}
	}
}
```

# Get Url without Query
```java
private String getUrlWithoutParameters(String url) throws URISyntaxException {
    URI uri = new URI(url);
    return new URI(uri.getScheme(),
		   uri.getAuthority(),
		   uri.getPath(),
		   null, // Ignore the query part of the input url
		   uri.getFragment()).toString();
}
````
